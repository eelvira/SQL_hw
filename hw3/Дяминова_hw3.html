temp1 = %sql (select t.customer_id, sum(t.list_price), max(t.list_price), min(t.list_price), count(t.transaction_id)\

temp2 = %sql (select distinct t.customer_id \
        , sum(t.list_price) OVER (PARTITION BY t.customer_id)\
        , max(t.list_price) OVER (PARTITION BY t.customer_id)\
        , min(t.list_price) OVER (PARTITION BY t.customer_id)\
        , count(t.transaction_id) OVER (PARTITION BY t.customer_id)\

Количество несоответсвующих строк: 1162

Может показаться, что это странный ответ, учитывая, что визуально строки похожи. Однако это связано с тем, что в этих запросах суммирование была над float значениями. А это приближенное значение, соответственно, возможна небольшая ошибка при вычислениях. Для того, чтобы избежать таких случаев, можно числа переводить в decimal. Ниже пример запросов, которые показывают результат запросов с decimal. В результате мы получаем одинаковые запросы.

%%sql


create temp table max_interval as
with cte as (
    select t.customer_id
    ,max(to_date(t.transaction_date, 'DD/MM/YYYY')) - min(to_date(t.transaction_date, 'DD/MM/YYYY')) as day_difference
    from transaction_20240101 t
    group by t.customer_id)

select customer_id, day_difference
from cte
where day_difference = (select max(day_difference) from cte);


select m.customer_id, c.first_name, c.last_name, c.job_title
from customer_20240101 c
join max_interval m
    on m.customer_id = c.customer_id;
